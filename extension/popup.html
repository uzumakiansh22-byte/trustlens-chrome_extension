function renderResult() { chrome.storage.local.get("result", (data) => { const output = document.getElementById("output"); if (!data.result) return; if (data.result.loading) { output.innerHTML = `
⏳ Analyzing with AI...
`; return; } if (data.result.error) { output.innerHTML = `
Error: ${data.result.error}
`; return; } // Handle results array for text or single result for image/video const results = data.result.results || [data.result.result]; output.innerHTML = results.map(r => `
${r.trust_score}%

${r.verdict}

${r.summary || r.reason}

Flags: ${(r.key_flags || []).join(", ")}
`).join(""); }); } // Watch for changes while the side panel is open chrome.storage.onChanged.addListener(renderResult); // Render immediately when opened document.addEventListener("DOMContentLoaded", renderResult);
